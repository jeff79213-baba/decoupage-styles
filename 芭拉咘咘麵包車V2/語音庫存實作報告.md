```javascript
// ============ VOICE INPUT (GROQ) ============
const GROQ_API_URL = "https://api.groq.com/openai/v1/chat/completions";
const GROQ_MODEL = "llama-3.3-70b-versatile";

function getGroqKey() {
  return localStorage.getItem("bala_groq_key") || "";
}

function saveGroqKey(key) {
  localStorage.setItem("bala_groq_key", key);
}

let voiceLongPressTimer = null;

function showGroqKeyInput() {
  const modal = document.getElementById("voiceModal");
  const content = document.getElementById("voiceModalContent");
  const currentKey = getGroqKey();
  content.innerHTML = `
    <h2>🔑 GROQ API 金鑰</h2>
    <div style="margin:14px 0;font-size:13px;color:#888">
      請輸入你的 GROQ API Key<br>
      （不會顯示在設定頁面）
    </div>
    <input id="groqKeyInput" type="password" value="${currentKey}"
      style="width:100%;padding:10px;border:1.5px solid #ddd;border-radius:8px;font-size:14px;box-sizing:border-box"
      placeholder="gsk_...">
    <div style="display:flex;gap:8px;margin-top:14px">
      <button onclick="saveGroqKey(document.getElementById('groqKeyInput').value.trim());document.getElementById('voiceModal').classList.remove('open');showToast('API 金鑰已儲存')"
        style="flex:1;padding:10px;border:none;border-radius:8px;background:var(--primary);color:#fff;cursor:pointer;font-size:14px">儲存</button>
      <button onclick="document.getElementById('voiceModal').classList.remove('open')"
        style="flex:1;padding:10px;border:1px solid #ddd;border-radius:8px;background:var(--card);cursor:pointer;font-size:14px">取消</button>

  `;

async function startVoiceInput() {
  // 檢查 API Key
  if (!getGroqKey()) {
    showGroqKeyInput();
    return;
  }

  // 語音辨識
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  if (!SpeechRecognition) {
    showToast("您的瀏覽器不支援語音辨識，請使用 Chrome");
    return;
  }

  const btn = document.getElementById("voiceBtn");
  btn.classList.add("listening");

  const recognition = new SpeechRecognition();
  recognition.lang = "zh-TW";
  recognition.continuous = false;
  recognition.interimResults = false;

  recognition.onresult = async (event) => {
    btn.classList.remove("listening");
    const text = event.results[0][0].transcript;
    await processVoiceText(text);
  };

  recognition.onerror = (event) => {
    btn.classList.remove("listening");
    showToast("語音辨識失敗：" + event.error);
  };

  recognition.onend = () => {
    btn.classList.remove("listening");
  };

  // 顯示語中狀態 + 範例提示 + 打字備用
  const modal = document.getElementById("voiceModal");
  const content = document.getElementById("voiceModalContent");
  content.innerHTML = `
    <div style="text-align:center;padding:20px">
      <div style="font-size:48px;margin-bottom:10px">🎤</div>
      <div style="font-size:16px;font-weight:600;margin-bottom:8px">正在聆聽...</div>
      <div class="voiceHint">
        <strong>💡 試試這樣說：</strong><br>
        「菠蘿20個、蔥花15個，紅豆今天沒做」<br>
        可以同時說價格：「菠蘿1個30、做了20個」<br>
        沒提到的品項維持原庫存
      </div>
      <div style="font-size:13px;color:#888">也可以直接打字：</div>
      <div style="display:flex;gap:8px;margin-top:8px">
        <input id="voiceTextInput" style="flex:1;padding:10px;border:1.5px solid #ddd;border-radius:8px;font-size:14px" placeholder="例如：菠蘿20 蔥花15">
        <button onclick="processVoiceText(document.getElementById('voiceTextInput').value.trim())" style="padding:10px 16px;border:none;border-radius:8px;background:var(--primary);color:#fff;cursor:pointer;font-size:14px">送出</button>
      </div>
      <button onclick="document.getElementById('voiceModal').classList.remove('open')" style="margin-top:12px;padding:8px 20px;border:1px solid #ddd;border-radius:8px;background:var(--card);cursor:pointer;font-size:13px">取消</button>
    </div>
  `;
  modal.classList.add("open");

  recognition.start();
}

async function processVoiceText(text) {
  if (!text) { showToast("請輸入或說出內容"); return; }

  const modal = document.getElementById("voiceModal");
  const content = document.getElementById("voiceModalContent");

  // 顯示載入中
  content.innerHTML = `
    <div style="text-align:center;padding:20px">
      <div style="font-size:16px;margin-bottom:10px">🤔 AI 解析中...</div>
      <div style="font-size:13px;color:#888">「${text}」</div>
    </div>
  `;

  // 準備品項清單給 AI 參考
  const itemList = [];
  CATEGORIES.forEach(cat => {
    cat.items.forEach(item => {
      const priceStr = item.prices.map(p =>
        `${p.label}${item.unit||"個"}$${p.price}`
      ).join("、");
      itemList.push({ name: item.name, unit: item.unit||"個", prices: priceStr });
    });
  });

  try {
    const response = await fetch(GROQ_API_URL, {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": "Bearer " + getGroqKey()
      },
      body: JSON.stringify({
        model: GROQ_MODEL,
        messages: [
          {
            role: "system",
            content: `你是一個庫存管理助手。使用者的店家品項如下：
${JSON.stringify(itemList, null, 2)}

使用者的輸入是今天的烘焙數量，請解析並回傳 JSON（不要其他文字）：
{
  "items": [
    {
      "name": "品項完整名稱（必須與上述清單完全一致）",
      "stock": 數量（數字，如果說「沒做」就是0）,
      "price": 如果提到價格就放數字，沒提到就 null
    }
  ],
  "unmatched": ["無法對應到品項的文字"]
}

規則：
1. name 必須與上述品項清單完全一致，不可簡稱或改名
2. 如果使用者說「沒做」、「沒有」、「不用」等，stock 設為 0
3. 沒提到的品項不要放進 items 裡
4. 一個品項如果有多個價格，使用者提到的價格可能是其中一個；如果價格跟系統不同，以使用者說的為準`
          },
          { role: "user", content: text }
        ],
        temperature: 0.1,
        max_tokens: 1024
      })
    });

    const data = await response.json();
    const resultText = data.choices?.[0]?.message?.content || "";
    
    let result;
    try {
      result = JSON.parse(resultText.replace(/```json\s*/g, "").replace(/```\s*/g, ""));
    } catch(e) {
      showToast("AI 解析失敗，請再試一次");
      modal.classList.remove("open");
      return;
    }

    showVoiceResult(result);
  } catch(e) {
    console.error("GROQ 呼叫失敗", e);
    showToast("AI 連線失敗，請檢查 API 金鑰");
    modal.classList.remove("open");
  }
}

function showVoiceResult(result) {
  const modal = document.getElementById("voiceModal");
  const content = document.getElementById("voiceModalContent");

  // 建立對照表：品項名稱 → 完整品項資料
  const itemMap = {};
  CATEGORIES.forEach(cat => {
    cat.items.forEach(item => {
      itemMap[item.name] = item;
    });
  });

  let tableRows = "";
  let hasItems = false;

  result.items.forEach(item => {
    const sysItem = itemMap[item.name];
    if (!sysItem) return;
    hasItems = true;
    const priceDisplay = sysItem.prices.map(p =>
      `${p.label}${sysItem.unit||"個"} $${p.price}`
    ).join(" / ");
    const newPrice = item.price !== null && item.price !== undefined
      ? `（AI 辨識價格：$${item.price}）` : "";
    const statusText = item.stock === 0 ? "沒做" : "有做";
    const statusClass = item.stock === 0 ? "statusNo" : "statusYes";
    const stockValue = item.stock;

    tableRows += `<tr>
      <td><span class="itemName">${item.name}</span></td>
      <td><span class="itemPrice">${priceDisplay} ${newPrice}</span></td>
      <td class="itemStock"><input type="number" min="0" id="vr_${item.name}" value="${stockValue}"></td>
      <td><span class="${statusClass}">${statusText}</span></td>
    </tr>`;
  });

  // 找不到的品項警告
  let skippedRows = "";
  result.items.forEach(item => {
    if (!itemMap[item.name]) {
      skippedRows += `<div style="font-size:12px;color:#e74c3c;margin:2px 0">⚠️ 找不到「${item.name}」</div>`;
    }
  });

  content.innerHTML = `
    <h2>📋 確認庫存內容</h2>
    <div style="font-size:13px;color:#888;margin-bottom:8px">
      以下為 AI 辨識結果，請確認無誤後按「套用」
    </div>
    ${skippedRows}
    ${hasItems ? `
    <table class="voiceResultTable">
      <tr>
        <th>品項</th>
        <th>價格</th>
        <th>庫存</th>
        <th>狀態</th>
      </tr>
      ${tableRows}
    </table>
    ` : '<div style="text-align:center;padding:20px;color:#888">⚠️ 沒有辨識到任何品項</div>'}
    <div style="font-size:12px;color:#999;margin-top:6px">沒列到的品項保持原庫存不變</div>
    <div style="display:flex;gap:8px;margin-top:14px">
      <button onclick="execVoiceSave()" style="flex:1;padding:10px;border:none;border-radius:8px;background:var(--primary);color:#fff;cursor:pointer;font-size:14px">✔ 套用</button>
      <button onclick="document.getElementById('voiceModal').classList.remove('open')" style="flex:1;padding:10px;border:1px solid #ddd;border-radius:8px;background:var(--card);cursor:pointer;font-size:14px">✕ 取消</button>
    </div>
  `;

  // 儲存結果供 execVoiceSave 使用
  window._voiceResult = result;
}

function execVoiceSave() {
  const result = window._voiceResult;
  if (!result) return;

  let changed = 0;
  result.items.forEach(item => {
    const input = document.getElementById("vr_" + item.name);
    if (!input) return;
    const val = input.value.trim();
    if (val !== "") {
      const num = parseInt(val);
      if (!isNaN(num) && num >= 0) {
        STOCK[item.name] = { stock: num };
        changed++;
      }
    }
  });

  saveStockToFirebase();
  document.getElementById("voiceModal").classList.remove("open");
  window._voiceResult = null;

  if (document.getElementById("view-menu").classList.contains("active")) renderMenu();
  if (document.getElementById("view-admin").classList.contains("active")) renderAdminItems();
  showToast(`已更新 ${changed} 個品項的庫存`);
}
```
